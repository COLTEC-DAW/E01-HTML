                    //Temos 3 tipos de loop

                    //while
                    int a = 1;

                    while(a < 10){
                        printf("%d", &a);
                        a++;
                    }

                    //for
                    for(int i = 1; i < 10; i++){
                        printf("%d", &i);
                    }

                    //Do while
                    int i = 1;
                    do{
                        printf("%d", &i);
                        i++;
                    }while(i < 10);
